function validateForm(data) {
    const errors = [];
    
    if (!data.email.includes('@')) {
        errors.push('Некорректный email');
        highlightElement('email-error');
    }
    
    if (data.password.length < 8) {
        errors.push('Пароль слишком короткий');
        highlightElement('password-error');
    }
    
    return errors.length === 0;
}

// Плавная анимация по скроллу
function animateOnScroll() {
    const elements = document.querySelectorAll('.animate');
    
    elements.forEach(element => {
        const position = element.getBoundingClientRect();
        
        if (position.top < window.innerHeight * 0.8) {
            element.classList.add('animated');
            element.style.transform = 'translateY(0)';
            element.style.opacity = '1';
        }
    });
}

// Анимация чисел
function animateCounter(element, target) {
    let current = 0;
    const increment = target / 50;
    
    const timer = setInterval(() => {
        current += increment;
        if (current >= target) {
            element.textContent = target;
            clearInterval(timer);
        } else {
            element.textContent = Math.floor(current);
        }
    }, 20);
}

// Обработка drag-and-drop
function setupDragAndDrop() {
    const draggables = document.querySelectorAll('[draggable="true"]');
    
    draggables.forEach(draggable => {
        draggable.addEventListener('dragstart', (e) => {
            e.dataTransfer.setData('text/plain', draggable.id);
            draggable.classList.add('dragging');
        });
        
        draggable.addEventListener('dragend', () => {
            draggable.classList.remove('dragging');
        });
    });
}

// Динамическое обновление контента
function updateContent(data) {
    const container = document.getElementById('content');
    
    // Безопасное обновление DOM
    container.innerHTML = '';
    data.forEach(item => {
        const element = document.createElement('div');
        element.textContent = sanitizeInput(item.content);
        container.appendChild(element);
    });
    
    // Анимация появления
    animateElements(container.children);
}
